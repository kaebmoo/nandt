# 1. AvailabilityTemplate (‡πÄ‡∏û‡∏¥‡πà‡∏° capacity concept)
class AvailabilityTemplate(TenantBase):
    __tablename__ = 'availability_templates'
    
    id = Column(Integer, primary_key=True)
    name = Column(String(100), nullable=False)
    description = Column(Text)
    
    # üÜï Template Type
    template_type = Column(
        String(20), 
        default='dedicated'  # 'dedicated', 'shared', 'pool'
    )
    
    # üÜï Capacity Settings
    max_concurrent_slots = Column(Integer, default=1)  # ‡∏à‡∏≥‡∏ô‡∏ß‡∏ô slot ‡∏û‡∏£‡πâ‡∏≠‡∏°‡∏Å‡∏±‡∏ô‡∏™‡∏π‡∏á‡∏™‡∏∏‡∏î
    requires_provider_assignment = Column(Boolean, default=True)  # ‡∏ï‡πâ‡∏≠‡∏á‡∏£‡∏∞‡∏ö‡∏∏‡∏ß‡πà‡∏≤‡πÉ‡∏Ñ‡∏£‡∏°‡∏≤‡∏ó‡∏≥‡∏á‡∏≤‡∏ô‡πÑ‡∏´‡∏°
    
    # Existing fields...
    timezone = Column(String(50), default='Asia/Bangkok')
    is_active = Column(Boolean, default=True)
    created_at = Column(DateTime, default=lambda: datetime.datetime.now(datetime.timezone.utc))
    updated_at = Column(DateTime, onupdate=lambda: datetime.datetime.now(datetime.timezone.utc))
    
    # Relationships
    availabilities = relationship("Availability", back_populates="template")
    date_overrides = relationship("DateOverride", back_populates="template")
    event_types = relationship("EventType", back_populates="availability_template")
    
    # üÜï Provider assignments
    template_providers = relationship("TemplateProvider", back_populates="template")
    provider_schedules = relationship("ProviderSchedule", back_populates="template")


# 2. ‡∏ï‡∏≤‡∏£‡∏≤‡∏á‡∏Å‡∏•‡∏≤‡∏á: Template <-> Provider (many-to-many)
class TemplateProvider(TenantBase):
    """‡∏Å‡∏≥‡∏´‡∏ô‡∏î‡∏ß‡πà‡∏≤ provider ‡πÑ‡∏´‡∏ô‡πÉ‡∏ä‡πâ template ‡πÑ‡∏´‡∏ô‡πÑ‡∏î‡πâ‡∏ö‡πâ‡∏≤‡∏á"""
    __tablename__ = 'template_providers'
    
    id = Column(Integer, primary_key=True)
    template_id = Column(Integer, ForeignKey('availability_templates.id', ondelete='CASCADE'))
    provider_id = Column(Integer, ForeignKey('providers.id', ondelete='CASCADE'))
    
    # Settings
    is_primary = Column(Boolean, default=False)  # provider ‡∏´‡∏•‡∏±‡∏Å‡∏Ç‡∏≠‡∏á template ‡∏ô‡∏µ‡πâ
    can_auto_assign = Column(Boolean, default=True)  # ‡πÉ‡∏ä‡πâ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö auto-assignment ‡πÑ‡∏î‡πâ‡πÑ‡∏´‡∏°
    priority = Column(Integer, default=0)  # ‡∏•‡∏≥‡∏î‡∏±‡∏ö‡∏Ñ‡∏ß‡∏≤‡∏°‡∏™‡∏≥‡∏Ñ‡∏±‡∏ç (‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö auto-assign)
    
    created_at = Column(DateTime, default=lambda: datetime.datetime.now(datetime.timezone.utc))
    
    # Relationships
    template = relationship("AvailabilityTemplate", back_populates="template_providers")
    provider = relationship("Provider", back_populates="template_assignments")
    
    __table_args__ = (
        UniqueConstraint('template_id', 'provider_id', name='uq_template_provider'),
    )


# 3. ‡∏Å‡∏≥‡∏´‡∏ô‡∏î‡∏Å‡∏≤‡∏£‡∏ó‡∏≥‡∏á‡∏≤‡∏ô‡∏à‡∏£‡∏¥‡∏á (Provider Schedule)

    __tablename__ = 'provider_schedules'

    provider_id = Column(Integer, ForeignKey('providers.id', ondelete='CASCADE'), nullable=False)
    template_id = Column(Integer, ForeignKey('availability_templates.id', ondelete='CASCADE'))
    
    # Date range
    effective_date = Column(Date, nullable=False)  # ‡∏ß‡∏±‡∏ô‡∏ó‡∏µ‡πà‡πÄ‡∏£‡∏¥‡πà‡∏°‡∏ï‡πâ‡∏ô
    end_date = Column(Date)  # ‡∏ñ‡πâ‡∏≤‡πÑ‡∏°‡πà‡∏£‡∏∞‡∏ö‡∏∏ = ‡πÉ‡∏ä‡πâ‡πÑ‡∏õ‡πÄ‡∏£‡∏∑‡πà‡∏≠‡∏¢‡πÜ
    
    # Recurrence (optional - ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö rotation)
    recurrence_pattern = Column(String(50))  # 'weekly', 'biweekly', 'monthly', null
    days_of_week = Column(JSON)  # [1, 3, 5] = ‡∏à‡∏±‡∏ô‡∏ó‡∏£‡πå, ‡∏û‡∏∏‡∏ò, ‡∏®‡∏∏‡∏Å‡∏£‡πå

    custom_start_time = Column(Time)  # ‡∏ñ‡πâ‡∏≤‡πÄ‡∏£‡∏¥‡πà‡∏°‡∏ï‡πà‡∏≤‡∏á‡∏à‡∏≤‡∏Å template
    custom_end_time = Column(Time)    # ‡∏ñ‡πâ‡∏≤‡∏à‡∏ö‡∏ï‡πà‡∏≤‡∏á‡∏à‡∏≤‡∏Å template
    
    # Status
    is_active = Column(Boolean, default=True)
    schedule_type = Column(String(20), default='regular')  # 'regular', 'on_call', 'temporary'
    
    notes = Column(Text)  # ‡∏ö‡∏±‡∏ô‡∏ó‡∏∂‡∏Å‡πÄ‡∏û‡∏¥‡πà‡∏°‡πÄ‡∏ï‡∏¥‡∏° (‡πÄ‡∏ä‡πà‡∏ô "‡∏õ‡∏µ‡∏ô‡∏µ‡πâ‡∏´‡∏°‡∏≠‡∏´‡∏¢‡∏∏‡∏î‡∏ó‡∏∏‡∏Å‡∏®‡∏∏‡∏Å‡∏£‡πå")
    
    created_at = Column(DateTime, default=lambda: datetime.datetime.now(datetime.timezone.utc))
    updated_at = Column(DateTime, onupdate=lambda: datetime.datetime.now(datetime.timezone.utc))
    created_by = Column(String(100))  # ‡πÉ‡∏Ñ‡∏£‡∏™‡∏£‡πâ‡∏≤‡∏á schedule ‡∏ô‡∏µ‡πâ

## üîç ‡∏Å‡∏≤‡∏£‡∏Ñ‡∏≥‡∏ô‡∏ß‡∏ì Available Slots (Core Logic)

```python
def get_available_slots(template_id, target_date, target_time):
    """
    ‡∏Ñ‡∏≥‡∏ô‡∏ß‡∏ì‡∏ß‡πà‡∏≤ slot ‡∏ô‡∏±‡πâ‡∏ô‡∏ß‡πà‡∏≤‡∏á‡∏´‡∏£‡∏∑‡∏≠‡πÑ‡∏°‡πà
    
    Returns:
        {
            'is_available': True/False,
            'total_capacity': 5,
            'booked_count': 2,
            'available_count': 3,
            'available_providers': [provider1, provider2, provider3]
        }
    """
    
    # 1. ‡∏î‡∏∂‡∏á template
    template = db.query(AvailabilityTemplate).get(template_id)
    
    # 2. ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö DateOverride (‡∏ß‡∏±‡∏ô‡∏´‡∏¢‡∏∏‡∏î/‡∏õ‡∏¥‡∏î‡∏û‡∏¥‡πÄ‡∏®‡∏©)
    override = db.query(DateOverride).filter(
        DateOverride.template_id == template_id,
        DateOverride.date == target_date
    ).first()
    
    if override and override.is_unavailable:
        return {'is_available': False, 'reason': 'Date closed'}
    
    # 3. ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö Holiday
    holiday = db.query(Holiday).filter(
        Holiday.date == target_date,
        Holiday.is_active == True
    ).first()
    
    if holiday:
        return {'is_available': False, 'reason': 'Holiday'}
    
    # 4. ‡∏´‡∏≤ providers ‡∏ó‡∏µ‡πà‡∏ó‡∏≥‡∏á‡∏≤‡∏ô‡∏ß‡∏±‡∏ô‡∏ô‡∏µ‡πâ
    day_of_week = target_date.weekday()
    
    working_providers = db.query(ProviderSchedule).filter(
        ProviderSchedule.template_id == template_id,
        ProviderSchedule.effective_date <= target_date,
        or_(
            ProviderSchedule.end_date.is_(None),
            ProviderSchedule.end_date >= target_date
        ),
        ProviderSchedule.is_active == True,
        ProviderSchedule.days_of_week.contains([day_of_week])
    ).all()
    
    if not working_providers:
        return {'is_available': False, 'reason': 'No providers scheduled'}
    
    # 5. ‡∏Å‡∏£‡∏≠‡∏á providers ‡∏ó‡∏µ‡πà‡∏•‡∏≤
    provider_ids = [s.provider_id for s in working_providers]
    
    on_leave = db.query(ProviderLeave).filter(
        ProviderLeave.provider_id.in_(provider_ids),
        ProviderLeave.start_date <= target_date,
        ProviderLeave.end_date >= target_date,
        ProviderLeave.is_approved == True
    ).all()
    
    leave_provider_ids = [l.provider_id for l in on_leave]
    available_provider_ids = [p for p in provider_ids if p not in leave_provider_ids]
    
    if not available_provider_ids:
        return {'is_available': False, 'reason': 'All providers on leave'}
    
    # 6. ‡∏ï‡∏£‡∏ß‡∏à Resource Capacity (‡∏ñ‡πâ‡∏≤‡∏°‡∏µ)
    capacity = db.query(ResourceCapacity).filter(
        ResourceCapacity.template_id == template_id,
        or_(
            ResourceCapacity.specific_date == target_date,
            ResourceCapacity.day_of_week == DayOfWeek(day_of_week)
        ),
        ResourceCapacity.is_active == True
    ).first()
    
    max_concurrent = capacity.max_concurrent_appointments if capacity else template.max_concurrent_slots
    
    # 7. ‡∏ô‡∏±‡∏ö‡∏à‡∏≥‡∏ô‡∏ß‡∏ô‡∏ó‡∏µ‡πà‡∏à‡∏≠‡∏á‡πÅ‡∏•‡πâ‡∏ß
    target_datetime = datetime.combine(target_date, target_time)
    
    booked_count = db.query(Appointment).filter(
        Appointment.provider_id.in_(available_provider_ids),
        Appointment.start_time == target_datetime,
        Appointment.status.in_(['confirmed', 'pending'])
    ).count()
    
    # 8. ‡∏Ñ‡∏≥‡∏ô‡∏ß‡∏ì
    available_count = min(len(available_provider_ids), max_concurrent) - booked_count
    
    return {
        'is_available': available_count > 0,
        'total_capacity': max_concurrent,
        'booked_count': booked_count,
        'available_count': available_count,
        'available_providers': [
            p for p in available_provider_ids 
            if p not in already_booked_provider_ids
        ][:available_count]
    }
```
